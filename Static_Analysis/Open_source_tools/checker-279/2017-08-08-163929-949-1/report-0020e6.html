578 packetbuf_set_addr(PACKETBUF_ADDR_SENDER, &linkaddr_node_addr);
579#endif
580 if(packetbuf_holds_broadcast()) {
581 is_broadcast = 1;
582 PRINTDEBUG("contikimac: send broadcast\n");
583
584 if(broadcast_rate_drop()) {
585 return MAC_TX_COLLISION;
586 }
587 } else {
588#if NETSTACK_CONF_WITH_IPV61
589 PRINTDEBUG("contikimac: send unicast to %02x%02x:%02x%02x:%02x%02x:%02x%02x\n",
590 packetbuf_addr(PACKETBUF_ADDR_RECEIVER)->u8[0],
591 packetbuf_addr(PACKETBUF_ADDR_RECEIVER)->u8[1],
592 packetbuf_addr(PACKETBUF_ADDR_RECEIVER)->u8[2],
593 packetbuf_addr(PACKETBUF_ADDR_RECEIVER)->u8[3],
594 packetbuf_addr(PACKETBUF_ADDR_RECEIVER)->u8[4],
595 packetbuf_addr(PACKETBUF_ADDR_RECEIVER)->u8[5],
596 packetbuf_addr(PACKETBUF_ADDR_RECEIVER)->u8[6],
597 packetbuf_addr(PACKETBUF_ADDR_RECEIVER)->u8[7]);
598#else /* NETSTACK_CONF_WITH_IPV6 */
599 PRINTDEBUG("contikimac: send unicast to %u.%u\n",
600 packetbuf_addr(PACKETBUF_ADDR_RECEIVER)->u8[0],
601 packetbuf_addr(PACKETBUF_ADDR_RECEIVER)->u8[1]);
602#endif /* NETSTACK_CONF_WITH_IPV6 */
603 }
604
605 if(!packetbuf_attr(PACKETBUF_ATTR_IS_CREATED_AND_SECURED)) {
606 packetbuf_set_attr(PACKETBUF_ATTR_MAC_ACK, 1);
607 if(NETSTACK_FRAMERframer_802154.create() < 0) {
608 PRINTF("contikimac: framer failed\n");
609 return MAC_TX_ERR_FATAL;
610 }
611 }
612
613 transmit_len = packetbuf_totlen();
614 NETSTACK_RADIOieee_mode_driver.prepare(packetbuf_hdrptr(), transmit_len);
615
616 if(!is_broadcast && !is_receiver_awake) {
617#if WITH_PHASE_OPTIMIZATION0
618 ret = phase_wait(packetbuf_addr(PACKETBUF_ADDR_RECEIVER),
619 CYCLE_TIME(65536 / 8), GUARD_TIME10 * (2 * (65536 / 8192 + 65536 / 2000)) + (6 * (65536 / 8192
+ 65536 / 2000)),
620 mac_callback, mac_callback_ptr, buf_list);
621 if(ret == PHASE_DEFERRED) {
622 return MAC_TX_DEFERRED;
623 }
624 if(ret != PHASE_UNKNOWN) {
625 is_known_receiver = 1;
626 }
627#endif /* WITH_PHASE_OPTIMIZATION */
628 }
629
630
631
632 /* By setting we_are_sending to one, we ensure that the rtimer
633 powercycle interrupt do not interfere with us sending the packet. */
634 we_are_sending = 1;
635
636 /* If we have a pending packet in the radio, we should not send now,
637 because we will trash the received packet. Instead, we signal
638 that we have a collision, which lets the packet be received. This
639 packet will be retransmitted later by the MAC protocol
640 instread. */
641 if(NETSTACK_RADIOieee_mode_driver.receiving_packet() || NETSTACK_RADIOieee_mode_driver.pending_packet()) {
642 we_are_sending = 0;
643 PRINTF("contikimac: collision receiving %d, pending %d\n",
644 NETSTACK_RADIO.receiving_packet(), NETSTACK_RADIO.pending_packet());
645 return MAC_TX_COLLISION;
646 }
647
648 /* Switch off the radio to ensure that we didn't start sending while
649 the radio was doing a channel check. */
650 off();
651
652
653 strobes = 0;
654
655 /* Send a train of strobes until the receiver answers with an ACK. */
656 collisions = 0;
657
658 got_strobe_ack = 0;
659
660 /* Set contikimac_is_on to one to allow the on() and off() functions
661 to control the radio. We restore the old value of
662 contikimac_is_on when we are done. */
663 contikimac_was_on = contikimac_is_on;
664 contikimac_is_on = 1;
665
666#if !RDC_CONF_HARDWARE_CSMA0
667 /* Check if there are any transmissions by others. */
668 /* TODO: why does this give collisions before sending with the mc1322x? */
669 if(is_receiver_awake == 0) {
670 int i;
671 for(i = 0; i < CCA_COUNT_MAX_TX6; ++i) {
672 t0 = RTIMER_NOW()rtimer_arch_now();
673 on();
674#if CCA_CHECK_TIME65536 / 8192 > 0
675 while(RTIMER_CLOCK_LT(RTIMER_NOW(), t0 + CCA_CHECK_TIME)(((int32_t)(((rtimer_arch_now())) - ((t0 + 65536 / 8192)))) <
0)) { }
676#endif
677 if(NETSTACK_RADIOieee_mode_driver.channel_clear() == 0) {
678 collisions++;
679 off();
680 break;
681 }
682 off();
683 t0 = RTIMER_NOW()rtimer_arch_now();
684 while(RTIMER_CLOCK_LT(RTIMER_NOW(), t0 + CCA_SLEEP_TIME)(((int32_t)(((rtimer_arch_now())) - ((t0 + 65536 / 2000)))) <
0)) { }
685 }
686 }
687
688 if(collisions > 0) {
689 we_are_sending = 0;
690 off();
691 PRINTF("contikimac: collisions before sending\n");
692 contikimac_is_on = contikimac_was_on;
693 return MAC_TX_COLLISION;
694 }
695#endif /* RDC_CONF_HARDWARE_CSMA */
696
697#if !RDC_CONF_HARDWARE_ACK0
698 if(!is_broadcast) {
699 /* Turn radio on to receive expected unicast ack. Not necessary
700 with hardware ack detection, and may trigger an unnecessary cca
701 or rx cycle */
702 on();
703 }
704 seqno = packetbuf_attr(PACKETBUF_ATTR_MAC_SEQNO);
705#endif
706
707 watchdog_periodic();
708 t0 = RTIMER_NOW()rtimer_arch_now();
709 for(strobes = 0, collisions = 0;
710 got_strobe_ack == 0 && collisions == 0 &&
711 RTIMER_CLOCK_LT(RTIMER_NOW(), t0 + STROBE_TIME)(((int32_t)(((rtimer_arch_now())) - ((t0 + ((65536 / 8) + 2 *
(2 * (65536 / 8192 + 65536 / 2000))))))) < 0); strobes++) {
712
713 watchdog_periodic();
714
715 if(!is_broadcast && (is_receiver_awake || is_known_receiver) &&
716 !RTIMER_CLOCK_LT(RTIMER_NOW(), t0 + MAX_PHASE_STROBE_TIME)(((int32_t)(((rtimer_arch_now())) - ((t0 + 65536 / 60)))) <
0)) {
717 PRINTF("miss to %d\n", packetbuf_addr(PACKETBUF_ADDR_RECEIVER)->u8[0]);
718 break;
719 }
720
721#if !RDC_CONF_HARDWARE_ACK0
722 len = 0;
Value stored to 'len' is never read
723#endif
724
725 {
726 rtimer_clock_t wt;
727#if WITH_PHASE_OPTIMIZATION0
728 rtimer_clock_t txtime = RTIMER_NOW()rtimer_arch_now();
729#endif
730#if RDC_CONF_HARDWARE_ACK0
731 int ret = NETSTACK_RADIOieee_mode_driver.transmit(transmit_len);
732#else
733 NETSTACK_RADIOieee_mode_driver.transmit(transmit_len);
734#endif
735
736#if RDC_CONF_HARDWARE_ACK0
737 /* For radios that block in the transmit routine and detect the
738 ACK in hardware */
739 if(ret == RADIO_TX_OK) {
740 if(!is_broadcast) {
741 got_strobe_ack = 1;
742#if WITH_PHASE_OPTIMIZATION0
743 encounter_time = txtime;
744#endif
745 break;
746 }
747 } else if (ret == RADIO_TX_NOACK) {
748 } else if (ret == RADIO_TX_COLLISION) {
749 PRINTF("contikimac: collisions while sending\n");
750 collisions++;
751 }
752 wt = RTIMER_NOW()rtimer_arch_now();
753 while(RTIMER_CLOCK_LT(RTIMER_NOW(), wt + INTER_PACKET_INTERVAL)(((int32_t)(((rtimer_arch_now())) - ((wt + 65536 / 2500)))) <
0)) { }
754#else /* RDC_CONF_HARDWARE_ACK */
755 /* Wait for the ACK packet */
756 wt = RTIMER_NOW()rtimer_arch_now();
757 while(RTIMER_CLOCK_LT(RTIMER_NOW(), wt + INTER_PACKET_INTERVAL)(((int32_t)(((rtimer_arch_now())) - ((wt + 65536 / 2500)))) <
0)) { }
758
759 if(!is_broadcast && (NETSTACK_RADIOieee_mode_driver.receiving_packet() ||
760 NETSTACK_RADIOieee_mode_driver.pending_packet() ||
761 NETSTACK_RADIOieee_mode_driver.channel_clear() == 0)) {
762 uint8_t ackbuf[ACK_LEN3];
763 wt = RTIMER_NOW()rtimer_arch_now();
764 while(RTIMER_CLOCK_LT(RTIMER_NOW(), wt + AFTER_ACK_DETECTED_WAIT_TIME)(((int32_t)(((rtimer_arch_now())) - ((wt + 65536 / 1500)))) <
0)) { }
765
766 len = NETSTACK_RADIOieee_mode_driver.read(ackbuf, ACK_LEN3);
767 if(len == ACK_LEN3 && seqno == ackbuf[ACK_LEN3 - 1]) {
768 got_strobe_ack = 1;
769#if WITH_PHASE_OPTIMIZATION0
770 encounter_time = txtime;
771#endif
772 break;
773 } else {
774 PRINTF("contikimac: collisions while sending\n");
775 collisions++;
776 }
777 }
778#endif /* RDC_CONF_HARDWARE_ACK */
779 }
780 }
781
782 off();
783
784 PRINTF("contikimac: send (strobes=%u, len=%u, %s, %s), done\n", strobes,
785 packetbuf_totlen(),
786 got_strobe_ack ? "ack" : "no ack",
787 collisions ? "collision" : "no collision");
788
789#if CONTIKIMAC_CONF_COMPOWER
790 /* Accumulate the power consumption for the packet transmission. */
791 compower_accumulate(&current_packet);
792
793 /* Convert the accumulated power consumption for the transmitted
794 packet to packet attributes so that the higher levels can keep
795 track of the amount of energy spent on transmitting the
796 packet. */
797 compower_attrconv(&current_packet);
798
799 /* Clear the accumulated power consumption so that it is ready for
800 the next packet. */
801 compower_clear(&current_packet);
802#endif /* CONTIKIMAC_CONF_COMPOWER */
803
804 contikimac_is_on = contikimac_was_on;
805 we_are_sending = 0;
806
807 /* Determine the return value that we will return from the
808 function. We must pass this value to the phase module before we
809 return from the function. */
810 if(collisions > 0) {
811 ret = MAC_TX_COLLISION;
812 } else if(!is_broadcast && !got_strobe_ack) {
813 ret = MAC_TX_NOACK;
814 } else {
815 ret = MAC_TX_OK;
816 }
817
818#if WITH_PHASE_OPTIMIZATION0
819 if(is_known_receiver && got_strobe_ack) {
820 PRINTF("no miss %d wake-ups %d\n",
821 packetbuf_addr(PACKETBUF_ADDR_RECEIVER)->u8[0],
822 strobes);
823 }
824
825 if(!is_broadcast) {
826 if(collisions == 0 && is_receiver_awake == 0) {
827 phase_update(packetbuf_addr(PACKETBUF_ADDR_RECEIVER),
828 encounter_time, ret);
829 }
830 }
831#endif /* WITH_PHASE_OPTIMIZATION */
832
833 return ret;
834}
835/*---------------------------------------------------------------------------*/
836static void
837qsend_packet(mac_callback_t sent, void *ptr)
838{
839 int ret = send_packet(sent, ptr, NULL((void*)0), 0);
840 if(ret != MAC_TX_DEFERRED) {
841 mac_call_sent_callback(sent, ptr, ret, 1);
842 }
843}
844/*---------------------------------------------------------------------------*/
845static void
846qsend_list(mac_callback_t sent, void *ptr, struct rdc_buf_list *buf_list)
847{
848 struct rdc_buf_list *curr;
849 struct rdc_buf_list *next;
850 int ret;
851 int is_receiver_awake;
852 int pending;
853
854 if(buf_list == NULL((void*)0)) {
855 return;
856 }
857 /* Do not send during reception of a burst */
858 if(we_are_receiving_burst) {
859 /* Prepare the packetbuf for callback */
860 queuebuf_to_packetbuf(buf_list->buf);
861 /* Return COLLISION so the MAC may try again later */
862 mac_call_sent_callback(sent, ptr, MAC_TX_COLLISION, 1);
863 return;
864 }
865
866 /* Create and secure frames in advance */
867 curr = buf_list;
868 do {
869 next = list_item_next(curr);
870 queuebuf_to_packetbuf(curr->buf);
871 if(!packetbuf_attr(PACKETBUF_ATTR_IS_CREATED_AND_SECURED)) {
872 /* create and secure this frame */
873 if(next != NULL((void*)0)) {
874 packetbuf_set_attr(PACKETBUF_ATTR_PENDING, 1);
875 }
876#if !NETSTACK_CONF_BRIDGE_MODE
877 /* If NETSTACK_CONF_BRIDGE_MODE is set, assume PACKETBUF_ADDR_SENDER is already set. */
878 packetbuf_set_addr(PACKETBUF_ADDR_SENDER, &linkaddr_node_addr);
879#endif
880 packetbuf_set_attr(PACKETBUF_ATTR_MAC_ACK, 1);
881 if(NETSTACK_FRAMERframer_802154.create() < 0) {
882 PRINTF("contikimac: framer failed\n");
883 mac_call_sent_callback(sent, ptr, MAC_TX_ERR_FATAL, 1);
884 return;
885 }
886
887 packetbuf_set_attr(PACKETBUF_ATTR_IS_CREATED_AND_SECURED, 1);
888 queuebuf_update_from_packetbuf(curr->buf);
889 }
890 curr = next;
891 } while(next != NULL((void*)0));
892
893 /* The receiver needs to be awoken before we send */
894 is_receiver_awake = 0;
895 curr = buf_list;
896 do { /* A loop sending a burst of packets from buf_list */
897 next = list_item_next(curr);
898
899 /* Prepare the packetbuf */
900 queuebuf_to_packetbuf(curr->buf);
901
902 pending = packetbuf_attr(PACKETBUF_ATTR_PENDING);
903
904 /* Send the current packet */
905 ret = send_packet(sent, ptr, curr, is_receiver_awake);
906 if(ret != MAC_TX_DEFERRED) {
907 mac_call_sent_callback(sent, ptr, ret, 1);
908 }
909
910 if(ret == MAC_TX_OK) {
911 if(next != NULL((void*)0)) {
912 /* We're in a burst, no need to wake the receiver up again */
913 is_receiver_awake = 1;
914 curr = next;
915 }
916 } else {
917 /* The transmission failed, we stop the burst */
918 next = NULL((void*)0);
919 }
920 } while((next != NULL((void*)0)) && pending);
921}
922/*---------------------------------------------------------------------------*/
923/* Timer callback triggered when receiving a burst, after having
924 waited for a next packet for a too long time. Turns the radio off
925 and leaves burst reception mode */
926static void
927recv_burst_off(void *ptr)
928{
929 off();
930 we_are_receiving_burst = 0;
931}
932/*---------------------------------------------------------------------------*/
933static void
934input_packet(void)
935{
936 static struct ctimer ct;
937 int duplicate = 0;
938
939#if CONTIKIMAC_SEND_SW_ACK0
940 int original_datalen;
941 uint8_t *original_dataptr;
942
943 original_datalen = packetbuf_datalen();
944 original_dataptr = packetbuf_dataptr();
945#endif
946
947 if(!we_are_receiving_burst) {
948 off();
949 }
950
951 if(packetbuf_datalen() == ACK_LEN3) {
952 /* Ignore ack packets */
953 PRINTF("ContikiMAC: ignored ack\n");
954 return;
955 }
956
957 /* printf("cycle_start 0x%02x 0x%02x\n", cycle_start, cycle_start % CYCLE_TIME);*/
958
959 if(packetbuf_totlen() > 0 && NETSTACK_FRAMERframer_802154.parse() >= 0) {
960 if(packetbuf_datalen() > 0 &&
961 packetbuf_totlen() > 0 &&
962 (linkaddr_cmp(packetbuf_addr(PACKETBUF_ADDR_RECEIVER),
963 &linkaddr_node_addr) ||
964 packetbuf_holds_broadcast())) {
965 /* This is a regular packet that is destined to us or to the
966 broadcast address. */
967
968 /* If FRAME_PENDING is set, we are receiving a packets in a burst */
969 we_are_receiving_burst = packetbuf_attr(PACKETBUF_ATTR_PENDING);
970 if(we_are_receiving_burst) {
971 on();
972 /* Set a timer to turn the radio off in case we do not receive
973 a next packet */
974 ctimer_set(&ct, INTER_PACKET_DEADLINE128 / 32, recv_burst_off, NULL((void*)0));
975 } else {
976 off();
977 ctimer_stop(&ct);
978 }
979
980#if RDC_WITH_DUPLICATE_DETECTION!0
981 /* Check for duplicate packet. */
982 duplicate = mac_sequence_is_duplicate();
983 if(duplicate) {
984 /* Drop the packet. */
985 PRINTF("contikimac: Drop duplicate\n");
986 } else {
987 mac_sequence_register_seqno();
988 }
989#endif /* RDC_WITH_DUPLICATE_DETECTION */
990
991#if CONTIKIMAC_CONF_COMPOWER
992 /* Accumulate the power consumption for the packet reception. */
993 compower_accumulate(&current_packet);
994 /* Convert the accumulated power consumption for the received
995 packet to packet attributes so that the higher levels can
996 keep track of the amount of energy spent on receiving the
997 packet. */
998 compower_attrconv(&current_packet);
999
1000 /* Clear the accumulated power consumption so that it is ready
1001 for the next packet. */
1002 compower_clear(&current_packet);
1003#endif /* CONTIKIMAC_CONF_COMPOWER */
1004
1005 PRINTDEBUG("contikimac: data (%u)\n", packetbuf_datalen());
1006
1007#if CONTIKIMAC_SEND_SW_ACK0
1008 {
1009 frame802154_t info154;
1010 frame802154_parse(original_dataptr, original_datalen, &info154);
1011 if(info154.fcf.frame_type == FRAME802154_DATAFRAME(0x01) &&
1012 info154.fcf.ack_required != 0 &&
1013 linkaddr_cmp((linkaddr_t *)&info154.dest_addr,
1014 &linkaddr_node_addr)) {
1015 uint8_t ackdata[ACK_LEN3] = {0, 0, 0};
1016
1017 we_are_sending = 1;
1018 ackdata[0] = FRAME802154_ACKFRAME(0x02);
1019 ackdata[1] = 0;
1020 ackdata[2] = info154.seq;
1021 NETSTACK_RADIOieee_mode_driver.send(ackdata, ACK_LEN3);
1022 we_are_sending = 0;
1023 }
1024 }
1025#endif /* CONTIKIMAC_SEND_SW_ACK */
1026
1027 if(!duplicate) {
1028 NETSTACK_MACcsma_driver.input();
1029 }
1030 return;
1031 } else {
1032 PRINTDEBUG("contikimac: data not for us\n");
1033 }
1034 } else {
1035 PRINTF("contikimac: failed to parse (%u)\n", packetbuf_totlen());
1036 }
1037}
