363static void
364eventhandler(process_event_t ev, process_data_t data)
365{
366#if UIP_TCP(1)
367 unsigned char i;
368 register struct listenport *l;
369#endif /*UIP_TCP*/
370 struct process *p;
371
372 switch(ev) {
373 case PROCESS_EVENT_EXITED0x87:
374 /* This is the event we get if a process has exited. We go through
375 the TCP/IP tables to see if this process had any open
376 connections or listening TCP ports. If so, we'll close those
377 connections. */
378
379 p = (struct process *)data;
380#if UIP_TCP(1)
381 l = s.listenports;
382 for(i = 0; i < UIP_LISTENPORTS(40); ++i) {
383 if(l->p == p) {
384 uip_unlisten(l->port);
385 l->port = 0;
386 l->p = PROCESS_NONE((void*)0);
387 }
388 ++l;
389 }
390
391 {
392 struct uip_conn *cptr;
393
394 for(cptr = &uip_conns[0]; cptr < &uip_conns[UIP_CONNS(40)]; ++cptr) {
395 if(cptr->appstate.p == p) {
396 cptr->appstate.p = PROCESS_NONE((void*)0);
397 cptr->tcpstateflags = UIP_CLOSED0;
398 }
399 }
400 }
401#endif /* UIP_TCP */
402#if UIP_UDP1
403 {
404 struct uip_udp_conn *cptr;
405
406 for(cptr = &uip_udp_conns[0];
407 cptr < &uip_udp_conns[UIP_UDP_CONNS10]; ++cptr) {
408 if(cptr->appstate.p == p) {
409 cptr->lport = 0;
410 }
411 }
412 }
413#endif /* UIP_UDP */
414 break;
415
416 case PROCESS_EVENT_TIMER0x88:
417 /* We get this event if one of our timers have expired. */
418 {
419 /* Check the clock so see if we should call the periodic uIP
420 processing. */
421 if(data == &periodic &&
422 etimer_expired(&periodic)) {
423#if UIP_TCP(1)
424 for(i = 0; i < UIP_CONNS(40); ++i) {
425 if(uip_conn_active(i)(uip_conns[i].tcpstateflags != 0)) {
426 /* Only restart the timer if there are active
427 connections. */
428 etimer_restart(&periodic);
429 uip_periodic(i)do { uip_conn = &uip_conns[i]; uip_process(2); } while (0
);
430#if NETSTACK_CONF_WITH_IPV60
431 tcpip_ipv6_output();
432#else
433 if(uip_len > 0) {
434 PRINTF("tcpip_output from periodic len %d\n", uip_len);
435 tcpip_output();
436 PRINTF("tcpip_output after periodic len %d\n", uip_len);
437 }
438#endif /* NETSTACK_CONF_WITH_IPV6 */
439 }
440 }
441#endif /* UIP_TCP */
442#if UIP_CONF_IP_FORWARD0
443 uip_fw_periodic();
444#endif /* UIP_CONF_IP_FORWARD */
445 }
446
447#if NETSTACK_CONF_WITH_IPV60
448#if UIP_CONF_IPV6_REASSEMBLY0
449 /*
450 * check the timer for reassembly
451 */
452 if(data == &uip_reass_timer &&
453 etimer_expired(&uip_reass_timer)) {
454 uip_reass_over();
455 tcpip_ipv6_output();
456 }
457#endif /* UIP_CONF_IPV6_REASSEMBLY */
458 /*
459 * check the different timers for neighbor discovery and
460 * stateless autoconfiguration
461 */
462 /*if(data == &uip_ds6_timer_periodic &&
463 etimer_expired(&uip_ds6_timer_periodic)) {
464 uip_ds6_periodic();
465 tcpip_ipv6_output();
466 }*/
467#if !UIP_CONF_ROUTER1
468 if(data == &uip_ds6_timer_rs &&
469 etimer_expired(&uip_ds6_timer_rs)) {
470 uip_ds6_send_rs();
471 tcpip_ipv6_output();
472 }
473#endif /* !UIP_CONF_ROUTER */
474 if(data == &uip_ds6_timer_periodic &&
475 etimer_expired(&uip_ds6_timer_periodic)) {
476 uip_ds6_periodic();
477 tcpip_ipv6_output();
478 }
479#endif /* NETSTACK_CONF_WITH_IPV6 */
480 }
481 break;
482
483#if UIP_TCP(1)
484 case TCP_POLL:
485 if(data != NULL((void*)0)) {
486 uip_poll_conn(data)do { uip_conn = data; uip_process(3); } while (0);
487#if NETSTACK_CONF_WITH_IPV60
488 tcpip_ipv6_output();
489#else /* NETSTACK_CONF_WITH_IPV6 */
490 if(uip_len > 0) {
491 PRINTF("tcpip_output from tcp poll len %d\n", uip_len);
492 tcpip_output();
493 }
494#endif /* NETSTACK_CONF_WITH_IPV6 */
495 /* Start the periodic polling, if it isn't already active. */
496 start_periodic_tcp_timer();
497 }
498 break;
499#endif /* UIP_TCP */
500#if UIP_UDP1
501 case UDP_POLL:
502 if(data != NULL((void*)0)) {
503 uip_udp_periodic_conn(data)do { uip_udp_conn = data; uip_process(5); } while(0);
504#if NETSTACK_CONF_WITH_IPV60
505 tcpip_ipv6_output();
506#else
507 if(uip_len > 0) {
508 tcpip_output();
509 }
510#endif /* UIP_UDP */
511 }
512 break;
513#endif /* UIP_UDP */
514
515 case PACKET_INPUT:
516 packet_input();
517 break;
518 };
519}

528#if NETSTACK_CONF_WITH_IPV60
529void
530tcpip_ipv6_output(void)
531{
532 uip_ds6_nbr_t *nbr = NULL((void*)0);
533 uip_ipaddr_t *nexthop = NULL((void*)0);
534
535 if(uip_len == 0) {
536 return;
537 }
538
539 if(uip_len > UIP_LINK_MTU1280) {
540 UIP_LOG("tcpip_ipv6_output: Packet to big");
541 uip_clear_buf(){ uip_len = 0; };
542 return;
543 }
544
545 if(uip_is_addr_unspecified(&UIP_IP_BUF((struct uip_ip_hdr *)&(uip_aligned_buf.u8)[(14)])->destipaddr)){
546 UIP_LOG("tcpip_ipv6_output: Destination address unspecified");
547 uip_clear_buf(){ uip_len = 0; };
548 return;
549 }
550
551#if UIP_CONF_IPV6_RPL0
552 if(!rpl_update_header()) {
553 /* Packet can not be forwarded */
554 PRINTF("tcpip_ipv6_output: RPL header update error\n");
555 uip_clear_buf(){ uip_len = 0; };
556 return;
557 }
558#endif /* UIP_CONF_IPV6_RPL */
559
560 if(!uip_is_addr_mcast(&UIP_IP_BUF((struct uip_ip_hdr *)&(uip_aligned_buf.u8)[(14)])->destipaddr)) {
561 /* Next hop determination */
562
563#if UIP_CONF_IPV6_RPL0 && RPL_WITH_NON_STORING
564 uip_ipaddr_t ipaddr;
565 /* Look for a RPL Source Route */
566 if(rpl_srh_get_next_hop(&ipaddr)) {
567 nexthop = &ipaddr;
568 }
569#endif /* UIP_CONF_IPV6_RPL && RPL_WITH_NON_STORING */
570
571 nbr = NULL((void*)0);
572
573 /* We first check if the destination address is on our immediate
574 link. If so, we simply use the destination address as our
575 nexthop address. */
576 if(nexthop == NULL((void*)0) && uip_ds6_is_addr_onlink(&UIP_IP_BUF((struct uip_ip_hdr *)&(uip_aligned_buf.u8)[(14)])->destipaddr)){
577 nexthop = &UIP_IP_BUF((struct uip_ip_hdr *)&(uip_aligned_buf.u8)[(14)])->destipaddr;
578 }
579
580 if(nexthop == NULL((void*)0)) {
581 uip_ds6_route_t *route;
582 /* Check if we have a route to the destination address. */
583 route = uip_ds6_route_lookup(&UIP_IP_BUF((struct uip_ip_hdr *)&(uip_aligned_buf.u8)[(14)])->destipaddr);
584
585 /* No route was found - we send to the default route instead. */
586 if(route == NULL((void*)0)) {
587 PRINTF("tcpip_ipv6_output: no route found, using default route\n");
588 nexthop = uip_ds6_defrt_choose();
589 if(nexthop == NULL((void*)0)) {
590#ifdef UIP_FALLBACK_INTERFACE
591 PRINTF("FALLBACK: removing ext hdrs & setting proto %d %d\n",
592 uip_ext_len, *((uint8_t *)UIP_IP_BUF + 40));
593 if(uip_ext_len > 0) {
594 extern void remove_ext_hdr(void);
595 uint8_t proto = *((uint8_t *)UIP_IP_BUF((struct uip_ip_hdr *)&(uip_aligned_buf.u8)[(14)]) + 40);
596 remove_ext_hdr();
597 /* This should be copied from the ext header... */
598 UIP_IP_BUF((struct uip_ip_hdr *)&(uip_aligned_buf.u8)[(14)])->proto = proto;
599 }
600 /* Inform the other end that the destination is not reachable. If it's
601 * not informed routes might get lost unexpectedly until there's a need
602 * to send a new packet to the peer */
603 if(UIP_FALLBACK_INTERFACE.output() < 0) {
604 PRINTF("FALLBACK: output error. Reporting DST UNREACH\n");
605 uip_icmp6_error_output(ICMP6_DST_UNREACH, ICMP6_DST_UNREACH_ADDR, 0);
606 uip_flags = 0;
607 tcpip_ipv6_output();
608 return;
609 }
610#else
611 PRINTF("tcpip_ipv6_output: Destination off-link but no route\n");
612#endif /* !UIP_FALLBACK_INTERFACE */
613 uip_clear_buf(){ uip_len = 0; };
614 return;
615 }
616
617 } else {
618 /* A route was found, so we look up the nexthop neighbor for
619 the route. */
620 nexthop = uip_ds6_route_nexthop(route);
621
622 /* If the nexthop is dead, for example because the neighbor
623 never responded to link-layer acks, we drop its route. */
624 if(nexthop == NULL((void*)0)) {
625#if UIP_CONF_IPV6_RPL0
626 /* If we are running RPL, and if we are the root of the
627 network, we'll trigger a global repair berfore we remove
628 the route. */
629 rpl_dag_t *dag;
630 rpl_instance_t *instance;
631
632 dag = (rpl_dag_t *)route->state.dag;
633 if(dag != NULL((void*)0)) {
634 instance = dag->instance;
635
636 rpl_repair_root(instance->instance_id);
637 }
638#endif /* UIP_CONF_IPV6_RPL */
639 uip_ds6_route_rm(route);
640
641 /* We don't have a nexthop to send the packet to, so we drop
642 it. */
643 return;
644 }
645 }
646#if TCPIP_CONF_ANNOTATE_TRANSMISSIONS
647 if(nexthop != NULL((void*)0)) {
648 static uint8_t annotate_last;
649 static uint8_t annotate_has_last = 0;
650
651 if(annotate_has_last) {
652 printf("#L %u 0; red\n", annotate_last);
653 }
654 printf("#L %u 1; red\n", nexthop->u8[sizeof(uip_ipaddr_t) - 1]);
655 annotate_last = nexthop->u8[sizeof(uip_ipaddr_t) - 1];
656 annotate_has_last = 1;
657 }
658#endif /* TCPIP_CONF_ANNOTATE_TRANSMISSIONS */
659 }
660
661 /* End of next hop determination */
662
