1243{
1244#if RPL_WITH_DAO_ACK0
1245

1252
1253 buffer = UIP_ICMP_PAYLOAD((unsigned char *)&(uip_aligned_buf.u8)[((0) + 40 + uip_ext_len
+ 4)]);
1254

1256 sequence = buffer[2];
1257 status = buffer[3];
1258
1259 instance = rpl_get_instance(instance_id);
1260 if(instance == NULL((void*)0)) {
1261 uip_clear_buf(){ uip_len = 0; uip_ext_len = 0; };
1262 return;
1263 }
1264
1265 if(RPL_IS_STORING(instance)((1 != 1) && ((instance) != ((void*)0)) && ((
instance)->mop > 1))) {
1266 parent = rpl_find_parent(instance->current_dag, &UIP_IP_BUF((struct uip_ip_hdr *)&(uip_aligned_buf.u8)[(0)])->srcipaddr);
1267 if(parent == NULL((void*)0)) {
1268 /* not a known instance - drop the packet and ignore */
1269 uip_clear_buf(){ uip_len = 0; uip_ext_len = 0; };
1270 return;
1271 }
1272 } else {
1273 parent = NULL((void*)0);
1274 }
1275
1276 PRINTF("RPL: Received a DAO %s with sequence number %d (%d) and status %d from ",
1277 status < 128 ? "ACK" : "NACK",
1278 sequence, instance->my_dao_seqno, status);
1279 PRINT6ADDR(&UIP_IP_BUF->srcipaddr);
1280 PRINTF("\n");
1281
1282 if(sequence == instance->my_dao_seqno) {
1283 instance->has_downward_route = status < 128;
1284
1285 /* always stop the retransmit timer when the ACK arrived */
1286 ctimer_stop(&instance->dao_retransmit_timer);
1287
1288 /* Inform objective function on status of the DAO ACK */
1289 if(RPL_IS_STORING(instance)((1 != 1) && ((instance) != ((void*)0)) && ((
instance)->mop > 1)) && instance->of->dao_ack_callback) {
1290 instance->of->dao_ack_callback(parent, status);
1291 }
1292
1293#if RPL_REPAIR_ON_DAO_NACK0
1294 if(status >= RPL_DAO_ACK_UNABLE_TO_ACCEPT128) {
1295 /*
1296 * Failed the DAO transmission - need to remove the default route.
1297 * Trigger a local repair since we can not get our DAO in.
1298 */
1299 rpl_local_repair(instance);
1300 }
1301#endif
1302
1303 } else if(RPL_IS_STORING(instance)((1 != 1) && ((instance) != ((void*)0)) && ((
instance)->mop > 1))) {
1304 /* this DAO ACK should be forwarded to another recently registered route */
1305 uip_ds6_route_t *re;
1306 uip_ipaddr_t *nexthop;
1307 if((re = find_route_entry_by_dao_ack(sequence)) != NULL((void*)0)) {
1308 /* pick the recorded seq no from that node and forward DAO ACK - and
1309 clear the pending flag*/
1310 RPL_ROUTE_CLEAR_DAO_PENDING(re)do { (re)->state.state_flags &= ~0x02; } while(0);
1311
1312 nexthop = uip_ds6_route_nexthop(re);
1313 if(nexthop == NULL((void*)0)) {
1314 PRINTF("RPL: No next hop to fwd DAO ACK to\n");
1315 } else {
1316 PRINTF("RPL: Fwd DAO ACK to:");
1317 PRINT6ADDR(nexthop);
1318 PRINTF("\n");
1319 buffer[2] = re->state.dao_seqno_in;
1320 uip_icmp6_send(nexthop, ICMP6_RPL155, RPL_CODE_DAO_ACK0x03, 4);
1321 }
1322
1323 if(status >= RPL_DAO_ACK_UNABLE_TO_ACCEPT128) {
1324 /* this node did not get in to the routing tables above... - remove */
1325 uip_ds6_route_rm(re);
1326 }
1327 } else {
1328 PRINTF("RPL: No route entry found to forward DAO ACK (seqno %u)\n", sequence);
1329 }
1330 }
1331#endif /* RPL_WITH_DAO_ACK */
1332 uip_clear_buf(){ uip_len = 0; uip_ext_len = 0; };
1333}
