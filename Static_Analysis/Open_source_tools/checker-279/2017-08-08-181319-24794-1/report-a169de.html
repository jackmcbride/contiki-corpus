561/*---------------------------------------------------------------------------*/
562uint8_t
563get_match_length(uip_ipaddr_t *src, uip_ipaddr_t *dst)
564{
565 uint8_t j, k, x_or;
566 uint8_t len = 0;
567
568 for(j = 0; j < 16; j++) {
569 if(src->u8[j] == dst->u8[j]) {
570 len += 8;
571 } else {
572 x_or = src->u8[j] ^ dst->u8[j];
573 for(k = 0; k < 8; k++) {
574 if((x_or & 0x80) == 0) {
575 len++;
576 x_or <<= 1;
577 } else {
578 break;
579 }
580 }
581 break;
582 }
583 }
584 return len;
585}
