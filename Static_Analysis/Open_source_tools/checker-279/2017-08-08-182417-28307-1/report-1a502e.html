852static rpl_parent_t *
853best_parent(rpl_dag_t *dag, int fresh_only)
854{
855 rpl_parent_t *p;
856 rpl_of_t *of;
857 rpl_parent_t *best = NULL((void*)0);

865 for(p = nbr_table_head(rpl_parents); p != NULL((void*)0); p = nbr_table_next(rpl_parents, p)) {
866

868 if(p->dag != dag || p->rank == INFINITE_RANK0xffff || p->rank < ROOT_RANK(dag->instance)(dag->instance)->min_hoprankinc) {
869 if(p->rank < ROOT_RANK(dag->instance)(dag->instance)->min_hoprankinc) {
870 PRINTF("RPL: Parent has invalid rank\n");

880#if UIP_ND6_SEND_NS(1 && !1)

899{

1340{

1342
1343 /*
1344 * We recalculate ranks when we receive feedback from the system rather
1345 * than RPL protocol messages. This periodical recalculation is called
1346 * from a timer in order to keep the stack depth reasonably low.
1347 */
1348 p = nbr_table_head(rpl_parents);
1349 while(p != NULL((void*)0)) {
1350 if(p->dag != NULL((void*)0) && p->dag->instance && (p->flags & RPL_PARENT_FLAG_UPDATED0x1)) {
1351 p->flags &= ~RPL_PARENT_FLAG_UPDATED0x1;
1352 PRINTF("RPL: rpl_process_parent_event recalculate_ranks\n");

1354 PRINTF("RPL: A parent was dropped\n");
1355 }
1356 }
1357 p = nbr_table_next(rpl_parents, p);

1435/*---------------------------------------------------------------------------*/
1436void
1437rpl_process_dio(uip_ipaddr_t *from, rpl_dio_t *dio)
1438{
1439 rpl_instance_t *instance;
1440 rpl_dag_t *dag, *previous_dag;
1441 rpl_parent_t *p;
1442
1443#if RPL_WITH_MULTICAST
1444 /* If the root is advertising MOP 2 but we support MOP 3 we can still join
1445 * In that scenario, we suppress DAOs for multicast targets */
1446 if(dio->mop < RPL_MOP_STORING_NO_MULTICAST2) {
1447#else
1448 if(dio->mop != RPL_MOP_DEFAULT2) {
1449#endif
1450 PRINTF("RPL: Ignoring a DIO with an unsupported MOP: %d\n", dio->mop);
1451 return;
1452 }
1453
1454 dag = get_dag(dio->instance_id, &dio->dag_id);
1455 instance = rpl_get_instance(dio->instance_id);
1456
1457 if(dag != NULL((void*)0) && instance != NULL((void*)0)) {
1458 if(lollipop_greater_than(dio->version, dag->version)) {
1459 if(dag->rank == ROOT_RANK(instance)(instance)->min_hoprankinc) {
1460 PRINTF("RPL: Root received inconsistent DIO version number (current: %u, received: %u)\n", dag->version, dio->version);
1461 dag->version = dio->version;
1462 RPL_LOLLIPOP_INCREMENT(dag->version)do { if((dag->version) > 127) { (dag->version) = ((dag
->version) + 1) & 255; } else { (dag->version) = ((
dag->version) + 1) & 127; } } while(0);
1463 rpl_reset_dio_timer(instance);
1464 } else {
1465 PRINTF("RPL: Global repair\n");
1466 if(dio->prefix_info.length != 0) {
1467 if(dio->prefix_info.flags & UIP_ND6_RA_FLAG_AUTONOMOUS0x40) {
1468 PRINTF("RPL: Prefix announced in DIO\n");
1469 rpl_set_prefix(dag, &dio->prefix_info.prefix, dio->prefix_info.length);
1470 }
1471 }
1472 global_repair(from, dag, dio);
1473 }
1474 return;
1475 }
1476
1477 if(lollipop_greater_than(dag->version, dio->version)) {
1478 /* The DIO sender is on an older version of the DAG. */
1479 PRINTF("RPL: old version received => inconsistency detected\n");
1480 if(dag->joined) {
1481 rpl_reset_dio_timer(instance);
1482 return;
1483 }
1484 }
1485 }
1486
1487 if(instance == NULL((void*)0)) {
1488 PRINTF("RPL: New instance detected (ID=%u): Joining...\n", dio->instance_id);
1489 if(add_nbr_from_dio(from, dio)) {
1490 rpl_join_instance(from, dio);
1491 } else {
1492 PRINTF("RPL: Not joining since could not add parent\n");
1493 }
1494 return;
1495 }
1496
1497 if(instance->current_dag->rank == ROOT_RANK(instance)(instance)->min_hoprankinc && instance->current_dag != dag) {
1498 PRINTF("RPL: Root ignored DIO for different DAG\n");
1499 return;
1500 }
1501
1502 if(dag == NULL((void*)0)) {
1503#if RPL_MAX_DAG_PER_INSTANCE2 > 1
1504 PRINTF("RPL: Adding new DAG to known instance.\n");
1505 if(!add_nbr_from_dio(from, dio)) {
1506 PRINTF("RPL: Could not add new DAG, could not add parent\n");
1507 return;
1508 }
1509 dag = rpl_add_dag(from, dio);
1510 if(dag == NULL((void*)0)) {
1511 PRINTF("RPL: Failed to add DAG.\n");
1512 return;
1513 }
1514#else /* RPL_MAX_DAG_PER_INSTANCE > 1 */
1515 PRINTF("RPL: Only one instance supported.\n");
1516 return;
1517#endif /* RPL_MAX_DAG_PER_INSTANCE > 1 */
1518 }
1519
1520
1521 if(dio->rank < ROOT_RANK(instance)(instance)->min_hoprankinc) {
1522 PRINTF("RPL: Ignoring DIO with too low rank: %u\n",
1523 (unsigned)dio->rank);
1524 return;
1525 }
1526
1527 /* Prefix Information Option treated to add new prefix */
1528 if(dio->prefix_info.length != 0) {
1529 if(dio->prefix_info.flags & UIP_ND6_RA_FLAG_AUTONOMOUS0x40) {
1530 PRINTF("RPL: Prefix announced in DIO\n");
1531 rpl_set_prefix(dag, &dio->prefix_info.prefix, dio->prefix_info.length);
1532 }
1533 }
1534
1535 if(!add_nbr_from_dio(from, dio)) {
1536 PRINTF("RPL: Could not add parent based on DIO\n");
1537 return;
1538 }
1539
1540 if(dag->rank == ROOT_RANK(instance)(instance)->min_hoprankinc) {
1541 if(dio->rank != INFINITE_RANK0xffff) {
1542 instance->dio_counter++;
1543 }
1544 return;
1545 }
1546
1547 /* The DIO comes from a valid DAG, we can refresh its lifetime */
1548 dag->lifetime = (1UL << (instance->dio_intmin + instance->dio_intdoubl)) * RPL_DAG_LIFETIME3 / 1000;
1549 PRINTF("Set dag ");
1550 PRINT6ADDR(&dag->dag_id);
1551 PRINTF(" lifetime to %ld\n", dag->lifetime);
1552
1553 /*
1554 * At this point, we know that this DIO pertains to a DAG that
1555 * we are already part of. We consider the sender of the DIO to be
1556 * a candidate parent, and let rpl_process_parent_event decide
1557 * whether to keep it in the set.
1558 */
1559
1560 p = rpl_find_parent(dag, from);
1561 if(p == NULL((void*)0)) {
1562 previous_dag = find_parent_dag(instance, from);
1563 if(previous_dag == NULL((void*)0)) {
1564 /* Add the DIO sender as a candidate parent. */
1565 p = rpl_add_parent(dag, dio, from);
1566 if(p == NULL((void*)0)) {
1567 PRINTF("RPL: Failed to add a new parent (");
1568 PRINT6ADDR(from);
1569 PRINTF(")\n");
1570 return;
1571 }
1572 PRINTF("RPL: New candidate parent with rank %u: ", (unsigned)p->rank);
1573 PRINT6ADDR(from);
1574 PRINTF("\n");
1575 } else {
1576 p = rpl_find_parent(previous_dag, from);
1577 if(p != NULL((void*)0)) {
1578 rpl_move_parent(previous_dag, dag, p);
1579 }
1580 }
1581 } else {
1582 if(p->rank == dio->rank) {
1583 PRINTF("RPL: Received consistent DIO\n");
1584 if(dag->joined) {
1585 instance->dio_counter++;
1586 }
1587 }
1588 }
1589 p->rank = dio->rank;
1590
1591 if(dio->rank == INFINITE_RANK0xffff && p == dag->preferred_parent) {
1592 /* Our preferred parent advertised an infinite rank, reset DIO timer */
1593 rpl_reset_dio_timer(instance);
1594 }
1595
1596 /* Parent info has been updated, trigger rank recalculation */
1597 p->flags |= RPL_PARENT_FLAG_UPDATED0x1;
1598
1599 PRINTF("RPL: preferred DAG ");
1600 PRINT6ADDR(&instance->current_dag->dag_id);
1601 PRINTF(", rank %u, min_rank %u, ",
1602 instance->current_dag->rank, instance->current_dag->min_rank);
1603 PRINTF("parent rank %u, link metric %u\n",
1604 p->rank, rpl_get_parent_link_metric(p));
1605
1606 /* We have allocated a candidate parent; process the DIO further. */
1607
1608#if RPL_WITH_MC0
1609 memcpy(&p->mc, &dio->mc, sizeof(p->mc))__builtin___memcpy_chk (&p->mc, &dio->mc, sizeof
(p->mc), __builtin_object_size (&p->mc, 0));
1610#endif /* RPL_WITH_MC */
1611 if(rpl_process_parent_event(instance, p) == 0) {
1612 PRINTF("RPL: The candidate parent is rejected\n");
1613 return;
1614 }
1615
1616 /* We don't use route control, so we can have only one official parent. */
1617 if(dag->joined && p == dag->preferred_parent) {
1618 if(should_refresh_routes(instance, dio, p)) {
1619 /* Our parent is requesting a new DAO. Increment DTSN in turn,
1620 * in both storing and non-storing mode (see RFC6550 section 9.6.) */
1621 RPL_LOLLIPOP_INCREMENT(instance->dtsn_out)do { if((instance->dtsn_out) > 127) { (instance->dtsn_out
) = ((instance->dtsn_out) + 1) & 255; } else { (instance
->dtsn_out) = ((instance->dtsn_out) + 1) & 127; } }
while(0);
1622 rpl_schedule_dao(instance);
1623 }
1624 /* We received a new DIO from our preferred parent.
1625 * Call uip_ds6_defrt_add to set a fresh value for the lifetime counter */
1626 uip_ds6_defrt_add(from, RPL_DEFAULT_ROUTE_INFINITE_LIFETIME1 ? 0 : RPL_LIFETIME(instance, instance->default_lifetime)(((instance->default_lifetime) == 0xFF) ? 0xFFFFFFFF : (unsigned
long)(instance)->lifetime_unit * (instance->default_lifetime
)));
1627 }
1628 p->dtsn = dio->dtsn;
1629}
