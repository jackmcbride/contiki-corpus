204reset(uint32_t uart_base)
205{
206 uint32_t lchr;
207
208 /* Make sure the UART is disabled before trying to configure it */
209 REG(uart_base + UART_CTL)(*((volatile unsigned long *)(uart_base + 0x00000030))) = UART_CTL_VALUE(0x00000200 | 0x00000100 | (0 << 5));
210
211 /* Clear error status */
212 REG(uart_base + UART_ECR)(*((volatile unsigned long *)(uart_base + 0x00000004))) = 0xFF;
213
214 /* Store LCHR configuration */
215 lchr = REG(uart_base + UART_LCRH)(*((volatile unsigned long *)(uart_base + 0x0000002C)));
216
217 /* Flush FIFOs by clearing LCHR.FEN */
218 REG(uart_base + UART_LCRH)(*((volatile unsigned long *)(uart_base + 0x0000002C))) = 0;
219
220 /* Restore LCHR configuration */
221 REG(uart_base + UART_LCRH)(*((volatile unsigned long *)(uart_base + 0x0000002C))) = lchr;
222
223 /* UART Enable */
224 REG(uart_base + UART_CTL)(*((volatile unsigned long *)(uart_base + 0x00000030))) |= UART_CTL_UARTEN0x00000001;
225}
