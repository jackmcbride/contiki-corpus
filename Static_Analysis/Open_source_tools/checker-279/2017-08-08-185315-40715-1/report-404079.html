304casecmp(char *str1, const char *str2, char len)
305{
306 static char c;
307
308 while(len > 0) {
309 c = *str1;
310 /* Force lower-case characters. */
311 if(c & 0x40) {
312 c |= 0x20;
313 }
314 if(*str2 != c) {
315 return 1;
316 }
317 ++str1;
318 ++str2;
319 --len;
320 }
