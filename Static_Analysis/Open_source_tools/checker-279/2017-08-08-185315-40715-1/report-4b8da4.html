78/* Write little endian 16 bit value */
79static void CC_INLINE__inline__
80le16_write(void *ptr, uint16_t v)
81{
82 uint8_t *p = (uint8_t *)ptr;
83 p[0] = v & 0xff;
84 p[1] = v >> 8;
85}
86/*---------------------------------------------------------------------------*/
87/* Read little endian 16 bit value */
88static uint16_t CC_INLINE__inline__
89le16_read(const void *ptr)
90{
91 const uint8_t *p = (const uint8_t *)ptr;
92 return ((uint16_t)p[1] << 8) | p[0];
93}
